  def statusParse(stat)
        case stat
          when :away : return "Zaraz wracam"
          when nil : return "Dostepny"
          when :chat : return "Pogadam"
          when :dnd : return "Nie przeszkadzac"
          when :xa : return "Nieosiagalny"
          when :unavailable : return "Niedostepny"
        end
        "Nieoczekiwany blad"
  end
