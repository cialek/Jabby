  def otrzymane_wiadomosci
        wiadomosci=[]
        while(!@kolejka_wiadomosci.empty?)
                wiadomosc=@kolejka_wiadomosci.pop(true) rescue nil
                break if wiadomosc.nil?
                wiadomosci << wiadomosc
                yield wiadomosc if block_given?
        end
        wiadomosci
  end
